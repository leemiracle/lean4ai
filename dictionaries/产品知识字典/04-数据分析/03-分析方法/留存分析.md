# 留存分析

## 1. 定义

留存分析是通过追踪用户在特定时间段内是否继续使用产品，评估产品粘性和用户忠诚度的分析方法。

### 核心概念

```
留存率 = (回访用户数 / 初始用户数) × 100%

关键指标：
├─ 次日留存（Day 1）
├─ 7日留存（Day 7）
├─ 30日留存（Day 30）
└─ 长期留存（Day 60/90/180）

留存分类：
├─ 短期留存：次日、3日、7日
├─ 中期留存：14日、30日
└─ 长期留存：60日、90日、180日
```

## 2. 留存分析方法

### 2.1 同期群留存分析

**SQL示例：同期群留存表**
```sql
-- 同期群留存分析（Cohort Retention）
WITH user_cohorts AS (
    -- 按注册月份分组
    SELECT 
        user_id,
        DATE_FORMAT(register_time, '%Y-%m') as cohort_month,
        DATE(register_time) as register_date
    FROM users
    WHERE register_time >= DATE_SUB(NOW(), INTERVAL 6 MONTH)
),
user_activity AS (
    -- 用户活跃数据
    SELECT 
        user_id,
        DATE(activity_time) as activity_date
    FROM user_activities
    WHERE activity_time >= DATE_SUB(NOW(), INTERVAL 6 MONTH)
    GROUP BY user_id, DATE(activity_time)
),
retention_matrix AS (
    SELECT 
        c.cohort_month,
        c.user_id,
        DATEDIFF(a.activity_date, c.register_date) as day_diff
    FROM user_cohorts c
    LEFT JOIN user_activity a ON c.user_id = a.user_id
)
SELECT 
    cohort_month,
    COUNT(DISTINCT user_id) as total_users,
    COUNT(DISTINCT CASE WHEN day_diff = 0 THEN user_id END) as day0,
    COUNT(DISTINCT CASE WHEN day_diff = 1 THEN user_id END) as day1,
    COUNT(DISTINCT CASE WHEN day_diff = 3 THEN user_id END) as day3,
    COUNT(DISTINCT CASE WHEN day_diff = 7 THEN user_id END) as day7,
    COUNT(DISTINCT CASE WHEN day_diff = 14 THEN user_id END) as day14,
    COUNT(DISTINCT CASE WHEN day_diff = 30 THEN user_id END) as day30,
    -- 留存率
    ROUND(COUNT(DISTINCT CASE WHEN day_diff = 1 THEN user_id END) * 100.0 
        / COUNT(DISTINCT user_id), 2) as day1_rate,
    ROUND(COUNT(DISTINCT CASE WHEN day_diff = 7 THEN user_id END) * 100.0 
        / COUNT(DISTINCT user_id), 2) as day7_rate,
    ROUND(COUNT(DISTINCT CASE WHEN day_diff = 30 THEN user_id END) * 100.0 
        / COUNT(DISTINCT user_id), 2) as day30_rate
FROM retention_matrix
GROUP BY cohort_month
ORDER BY cohort_month DESC;
```

### 2.2 留存曲线分析

**Python留存曲线绘制**
```python
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

def plot_retention_curve(retention_data):
    """
    绘制留存曲线
    """
    plt.figure(figsize=(12, 6))
    
    # 留存曲线
    days = retention_data['days']
    retention_rate = retention_data['retention_rate']
    
    plt.plot(days, retention_rate, marker='o', linewidth=2, markersize=8)
    
    # 添加参考线
    plt.axhline(y=retention_rate.iloc[-1], color='r', linestyle='--', alpha=0.5, 
                label=f'长期留存 {retention_rate.iloc[-1]:.1f}%')
    
    plt.xlabel('天数', fontsize=12)
    plt.ylabel('留存率 (%)', fontsize=12)
    plt.title('用户留存曲线', fontsize=16, fontweight='bold')
    plt.grid(True, alpha=0.3)
    plt.legend()
    
    plt.tight_layout()
    return plt

# 示例数据
retention_example = pd.DataFrame({
    'days': [0, 1, 3, 7, 14, 30, 60, 90],
    'retention_rate': [100, 45, 32, 28, 22, 18, 15, 13]
})

plot_retention_curve(retention_example)
plt.show()
```

## 3. 留存率计算方法

### 3.1 经典留存率计算

```sql
-- N日留存率计算
WITH 
-- 第0天用户
day0_users AS (
    SELECT 
        DATE(register_time) as cohort_date,
        user_id
    FROM users
    WHERE register_time >= DATE_SUB(NOW(), INTERVAL 30 DAY)
),
-- N日后活跃用户
retained_users AS (
    SELECT 
        DATE(u.register_time) as cohort_date,
        COUNT(DISTINCT u.user_id) as total_users,
        COUNT(DISTINCT CASE WHEN DATEDIFF(DATE(a.activity_time), DATE(u.register_time)) = 1 
            THEN u.user_id END) as day1_retained,
        COUNT(DISTINCT CASE WHEN DATEDIFF(DATE(a.activity_time), DATE(u.register_time)) = 7 
            THEN u.user_id END) as day7_retained,
        COUNT(DISTINCT CASE WHEN DATEDIFF(DATE(a.activity_time), DATE(u.register_time)) = 30 
            THEN u.user_id END) as day30_retained
    FROM users u
    LEFT JOIN user_activities a ON u.user_id = a.user_id
        AND a.activity_time >= u.register_time
        AND DATEDIFF(DATE(a.activity_time), DATE(u.register_time)) IN (1, 7, 30)
    WHERE u.register_time >= DATE_SUB(NOW(), INTERVAL 30 DAY)
    GROUP BY DATE(u.register_time)
)
SELECT 
    cohort_date,
    total_users,
    day1_retained,
    ROUND(day1_retained * 100.0 / total_users, 2) as day1_retention_rate,
    day7_retained,
    ROUND(day7_retained * 100.0 / total_users, 2) as day7_retention_rate,
    day30_retained,
    ROUND(day30_retained * 100.0 / total_users, 2) as day30_retention_rate
FROM retained_users
ORDER BY cohort_date DESC;
```

## 4. 留存影响因素分析

### 4.1 用户特征对留存的影响

```python
import pandas as pd
from scipy import stats

def analyze_retention_factors(user_data, retention_data):
    """
    分析影响留存的因素
    """
    results = []
    
    # 连续变量分析
    continuous_vars = ['age', 'income', 'usage_frequency']
    for var in continuous_vars:
        # 高留存组 vs 低留存组
        high_retention = user_data[user_data['retained'] == 1][var]
        low_retention = user_data[user_data['retained'] == 0][var]
        
        # t检验
        t_stat, p_value = stats.ttest_ind(high_retention, low_retention)
        
        results.append({
            'variable': var,
            'type': 'continuous',
            'high_retention_mean': high_retention.mean(),
            'low_retention_mean': low_retention.mean(),
            't_statistic': t_stat,
            'p_value': p_value,
            'significant': p_value < 0.05
        })
    
    # 分类变量分析
    categorical_vars = ['channel', 'device', 'region']
    for var in categorical_vars:
        # 卡方检验
        contingency_table = pd.crosstab(user_data[var], user_data['retained'])
        chi2, p_value, dof, expected = stats.chi2_contingency(contingency_table)
        
        results.append({
            'variable': var,
            'type': 'categorical',
            'chi2': chi2,
            'p_value': p_value,
            'significant': p_value < 0.05
        })
    
    return pd.DataFrame(results)
```

### 4.2 行为特征对留存的影响

```sql
-- 行为特征与留存的关系
WITH user_behaviors AS (
    SELECT 
        user_id,
        COUNT(*) as total_actions,
        COUNT(DISTINCT DATE(activity_time)) as active_days,
        AVG(session_duration) as avg_session_duration,
        COUNT(DISTINCT feature_used) as features_used,
        SUM(CASE WHEN is_social = 1 THEN 1 ELSE 0 END) as social_actions
    FROM user_activities
    WHERE activity_time >= DATE_SUB(NOW(), INTERVAL 7 DAY)
    GROUP BY user_id
),
user_retention AS (
    SELECT 
        user_id,
        MAX(CASE WHEN activity_time >= DATE_SUB(NOW(), INTERVAL 7 DAY) 
            AND activity_time < NOW() THEN 1 ELSE 0 END) as retained
    FROM user_activities
    WHERE activity_time >= DATE_SUB(NOW(), INTERVAL 14 DAY)
    GROUP BY user_id
)
SELECT 
    CASE 
        WHEN b.total_actions < 5 THEN '低活跃'
        WHEN b.total_actions < 20 THEN '中活跃'
        ELSE '高活跃'
    END as activity_level,
    COUNT(*) as total_users,
    SUM(r.retained) as retained_users,
    ROUND(AVG(r.retained) * 100, 2) as retention_rate,
    ROUND(AVG(b.active_days), 2) as avg_active_days,
    ROUND(AVG(b.features_used), 2) as avg_features_used
FROM user_behaviors b
JOIN user_retention r ON b.user_id = r.user_id
GROUP BY 
    CASE 
        WHEN b.total_actions < 5 THEN '低活跃'
        WHEN b.total_actions < 20 THEN '中活跃'
        ELSE '高活跃'
    END
ORDER BY retention_rate DESC;
```

## 6. 留存分析最佳实践

### 6.1 监控体系

```python
# 留存监控告警系统
class RetentionMonitor:
    def __init__(self, threshold_warning=0.15, threshold_critical=0.25):
        self.threshold_warning = threshold_warning
        self.threshold_critical = threshold_critical
    
    def check_retention_drop(self, current_rate, previous_rate):
        """
        检查留存率是否下降
        """
        drop_rate = (previous_rate - current_rate) / previous_rate
        
        if drop_rate >= self.threshold_critical:
            return {
                'status': 'critical',
                'message': f'留存率严重下降 {drop_rate*100:.1f}%',
                'action': '立即分析原因并采取措施'
            }
        elif drop_rate >= self.threshold_warning:
            return {
                'status': 'warning',
                'message': f'留存率下降 {drop_rate*100:.1f}%',
                'action': '密切关注并排查原因'
            }
        else:
            return {
                'status': 'normal',
                'message': '留存率正常',
                'action': '持续监控'
            }
```
